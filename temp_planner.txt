function EnhancedTripPlannerPage() {
    const [formData, setFormData] = useState({
        destination: '',
        interests: [],
        startDate: '',
        endDate: '',
        budget: 'medium',
        travelMode: 'train',
        experienceLevel: 'moderate'
    });
    const [loading, setLoading] = useState(false);
    const [itinerary, setItinerary] = useState(null);
    const [activeTab, setActiveTab] = useState('ai');
    const [suggestions, setSuggestions] = useState([]);
    const [aiInsights, setAiInsights] = useState(null);
    
    const generateItinerary = async () => {
        if (!formData.destination || formData.interests.length === 0) {
            alert('Please fill destination and select interests');
            return;
        }
        
        setLoading(true);
        
        try {
            const startDate = new Date(formData.startDate);
            const endDate = new Date(formData.endDate);
            const duration = Math.ceil((endDate - startDate) / (1000 * 60 * 60 * 24)) + 1;
            
            const res = await fetch(`${API_BASE}/trip/plan`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    destination: formData.destination,
                    interests: formData.interests,
                    duration,
                    budget: formData.budget,
                    travelMode: formData.travelMode,
                    plannerType: activeTab
                })
            });
            
            const data = await res.json();
            if (data.success) {
                setItinerary(data.itinerary);
                setAiInsights(data.insights);
            } else {
                alert('Failed to generate itinerary');
            }
        } catch (err) {
            console.error(err);
            alert('Error generating itinerary. Check backend connection.');
        } finally {
            setLoading(false);
        }
    };
    
    const toggleInterest = (interest) => {
        setFormData(prev => ({
            ...prev,
            interests: prev.interests.includes(interest)
                ? prev.interests.filter(i => i !== interest)
                : [...prev.interests, interest]
        }));
    };
    
    const handleViewOnMap = (location) => {
        localStorage.setItem('selectedLocation', JSON.stringify({
            name: location.name || location,
            type: 'destination'
        }));
        setCurrentPage('explore');
    };

    return (
        <div className="max-w-7xl mx-auto space-y-8 animate-fade-in p-6">
            <div className="flex justify-between items-center">
                <h2 className="text-4xl font-bold dark:text-white bg-gradient-to-r from-blue-600 to-indigo-600 bg-clip-text text-transparent">
                    AI Trip Planning Hub
                </h2>
                <div className="flex bg-gray-100 dark:bg-gray-700 rounded-lg p-1">
                    <button
                        onClick={() => setActiveTab('ai')}
                        className={`px-6 py-3 rounded-lg transition-all ${
                            activeTab === 'ai'
                                ? 'bg-gradient-to-r from-blue-500 to-indigo-600 text-white shadow-lg transform scale-105'
                                : 'text-gray-600 dark:text-gray-300 hover:bg-gray-200 dark:hover:bg-gray-600'
                        }`}
                    >
                        <span className="flex items-center gap-2">
                            <span role="img" aria-label="ai">🤖</span> AI Planner
                        </span>
                    </button>
                    <button
                        onClick={() => setActiveTab('smart')}
                        className={`px-6 py-3 rounded-lg transition-all ${
                            activeTab === 'smart'
                                ? 'bg-gradient-to-r from-purple-500 to-pink-600 text-white shadow-lg transform scale-105'
                                : 'text-gray-600 dark:text-gray-300 hover:bg-gray-200 dark:hover:bg-gray-600'
                        }`}
                    >
                        <span className="flex items-center gap-2">
                            <span role="img" aria-label="smart">✨</span> Smart Planner
                        </span>
                    </button>
                </div>
            </div>

            <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
                {/* Input Form */}
                <div className="bg-white dark:bg-gray-800 rounded-xl p-6 space-y-6 shadow-lg">
                    <div className="space-y-4">
                        <label className="block text-lg font-medium mb-2 dark:text-white">
                            {activeTab === 'ai' ? 'Where would you like to go?' : 'Enter your destinations'}
                        </label>
                        <input
                            type="text"
                            value={formData.destination}
                            onChange={(e) => setFormData({...formData, destination: e.target.value})}
                            className="w-full px-4 py-3 rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-700 text-gray-900 dark:text-white focus:ring-2 focus:ring-blue-500 dark:focus:ring-blue-400 focus:border-transparent transition-all"
                            placeholder={activeTab === 'ai' ? "e.g., Jaipur" : "e.g., Delhi, Agra, Jaipur"}
                        />
                    </div>
                    
                    {/* Travel Preferences */}
                    <div className="space-y-4">
                        <h3 className="text-lg font-medium dark:text-white">Travel Preferences</h3>
                        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                            {/* Budget */}
                            <div>
                                <label className="block text-sm text-gray-600 dark:text-gray-400 mb-2">Budget Range</label>
                                <select
                                    value={formData.budget}
                                    onChange={(e) => setFormData({...formData, budget: e.target.value})}
                                    className="w-full px-4 py-2 rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-700 text-gray-900 dark:text-white"
                                >
                                    <option value="budget">Budget</option>
                                    <option value="moderate">Moderate</option>
                                    <option value="luxury">Luxury</option>
                                </select>
                            </div>
                            
                            {/* Travel Mode */}
                            <div>
                                <label className="block text-sm text-gray-600 dark:text-gray-400 mb-2">Mode of Travel</label>
                                <select
                                    value={formData.travelMode}
                                    onChange={(e) => setFormData({...formData, travelMode: e.target.value})}
                                    className="w-full px-4 py-2 rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-700 text-gray-900 dark:text-white"
                                >
                                    <option value="train">Train</option>
                                    <option value="air">Flight</option>
                                    <option value="road">Road Trip</option>
                                </select>
                            </div>
                        </div>
                    </div>

                    {/* Interests */}
                    <div className="space-y-4">
                        <h3 className="text-lg font-medium dark:text-white">What interests you?</h3>
                        <div className="flex flex-wrap gap-3">
                            {[
                                { name: 'heritage', icon: '🏛️' },
                                { name: 'nature', icon: '🌿' },
                                { name: 'food', icon: '🍽️' },
                                { name: 'adventure', icon: '🏃' },
                                { name: 'wellness', icon: '🧘' },
                                { name: 'culture', icon: '🎭' },
                                { name: 'photography', icon: '📸' },
                                { name: 'shopping', icon: '🛍️' }
                            ].map(interest => (
                                <button
                                    key={interest.name}
                                    onClick={() => toggleInterest(interest.name)}
                                    className={`px-4 py-2 rounded-full capitalize transition-all transform hover:scale-105 ${
                                        formData.interests.includes(interest.name)
                                            ? 'bg-gradient-to-r from-blue-500 to-indigo-600 text-white shadow-lg'
                                            : 'bg-gray-100 dark:bg-gray-700 text-gray-700 dark:text-gray-300 hover:bg-gray-200 dark:hover:bg-gray-600'
                                    }`}
                                >
                                    <span className="mr-2">{interest.icon}</span>
                                    {interest.name}
                                </button>
                            ))}
                        </div>
                    </div>

                    {/* Generate Button */}
                    <button
                        onClick={generateItinerary}
                        disabled={loading}
                        className={`w-full py-4 rounded-xl text-white font-medium text-lg transition-all transform hover:scale-105 ${
                            loading
                                ? 'bg-gray-400 cursor-not-allowed'
                                : activeTab === 'ai'
                                    ? 'bg-gradient-to-r from-blue-500 to-indigo-600 hover:shadow-xl'
                                    : 'bg-gradient-to-r from-purple-500 to-pink-600 hover:shadow-xl'
                        }`}
                    >
                        {loading ? (
                            <span className="flex items-center justify-center gap-2">
                                <svg className="animate-spin h-5 w-5" viewBox="0 0 24 24">
                                    <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4" fill="none" />
                                    <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z" />
                                </svg>
                                Generating your perfect trip...
                            </span>
                        ) : (
                            <span className="flex items-center justify-center gap-2">
                                {activeTab === 'ai' ? '🤖' : '✨'} Generate {activeTab.toUpperCase()} Itinerary
                            </span>
                        )}
                    </button>
                </div>

                {/* Results Panel */}
                <div className="space-y-6">
                    {itinerary && (
                        <div className="bg-white dark:bg-gray-800 rounded-xl p-6 shadow-lg space-y-6 animate-fade-in-up">
                            <h3 className="text-2xl font-semibold dark:text-white flex items-center gap-3">
                                <span role="img" aria-label="itinerary">📍</span> Your Personalized Itinerary
                            </h3>
                            <div className="space-y-4">
                                {typeof itinerary === 'string' ? (
                                    <div className="prose dark:prose-invert max-w-none">
                                        {itinerary.split('\n').map((line, i) => (
                                            <p key={i} className="mb-2">{line}</p>
                                        ))}
                                    </div>
                                ) : (
                                    <div className="space-y-6">
                                        {itinerary.days.map((day, index) => (
                                            <div key={index} className="bg-gray-50 dark:bg-gray-700 rounded-lg p-4 space-y-3">
                                                <h4 className="text-lg font-medium text-blue-600 dark:text-blue-400">Day {index + 1}</h4>
                                                <div className="space-y-2">
                                                    {day.activities.map((activity, actIndex) => (
                                                        <div key={actIndex} className="flex items-start gap-3">
                                                            <span className="text-gray-400">•</span>
                                                            <p className="text-gray-700 dark:text-gray-300">{activity}</p>
                                                        </div>
                                                    ))}
                                                </div>
                                            </div>
                                        ))}
                                    </div>
                                )}
                            </div>
                        </div>
                    )}

                    {aiInsights && (
                        <div className="bg-white dark:bg-gray-800 rounded-xl p-6 shadow-lg space-y-4 animate-fade-in-up">
                            <h3 className="text-2xl font-semibold dark:text-white flex items-center gap-3">
                                <span role="img" aria-label="insights">💡</span> AI Travel Insights
                            </h3>
                            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                                {aiInsights.map((insight, index) => (
                                    <div key={index} className="bg-gray-50 dark:bg-gray-700 rounded-lg p-4">
                                        <h4 className="text-sm font-medium text-gray-500 dark:text-gray-400 mb-2">{insight.title}</h4>
                                        <p className="text-gray-800 dark:text-gray-200">{insight.description}</p>
                                    </div>
                                ))}
                            </div>
                            <button
                                onClick={() => handleViewOnMap(formData.destination)}
                                className="w-full mt-4 py-3 bg-gradient-to-r from-green-500 to-emerald-600 text-white rounded-lg flex items-center justify-center gap-2 hover:shadow-lg transition-all transform hover:scale-105"
                            >
                                <span role="img" aria-label="map">🗺️</span> View on Interactive Map
                            </button>
                        </div>
                    )}
                </div>
            </div>
        </div>
    );
}
